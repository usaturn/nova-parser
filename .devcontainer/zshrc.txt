function prompt_repo_name(){
    emulate -L zsh
    local repo_root
    repo_root="$(git rev-parse --show-toplevel 2>/dev/null)" || {
        print -r -- "Dev Container"
        return 0
    }
    repo_root="${repo_root:A}"
    print -r -- "${repo_root:t}"
}
repo_name=$(prompt_repo_name)

function genv(){

}
load_current_gcp_config() {
  local config_path="$HOME/.config/gcloud/active_config"
  if [ -f $config_path ]; then
    GCP_PROFILE=$(cat $config_path)
  fi
}
precmd() {
    psvar=()
    load_current_gcp_config
    psvar[1]=$GCP_PROFILE
}

function tmuxstart(){
    if [ -n "$TMUX" ]; then
        return 0
    fi
    project_dir=$(ls -1 /workspaces/)
    export TZ="${TZ:-Asia/Tokyo}"
    tmux -u new-session -A -s main -n main -c "/workspaces/${project_dir}" \; \
        set-environment -g TZ "${TZ}"
}

tmuxstart
