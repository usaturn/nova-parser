# For Git
function git-prompt {
  local branchname branch st remote pushed upstream
  branchname=`git symbolic-ref --short HEAD 2> /dev/null`
  if [ -z $branchname ]; then
    return
  fi
  st=`git status 2> /dev/null`
  if [[ -n `echo "$st" | grep "^nothing to"` ]]; then
    branch="%{${fg[green]}%}($branchname)%{$reset_color%}"
  elif [[ -n `echo "$st" | grep "^nothing added"` ]]; then
    branch="%{${fg[yellow]}%}($branchname)%{$reset_color%}"
  else
    branch="%{${fg[red]}%}($branchname)%{$reset_color%}"
  fi

  remote=`git config branch.${branchname}.remote 2> /dev/null`

  if [ -z $remote ]; then
    pushed=''
  else
    upstream="${remote}/${branchname}"
    if [[ -z `git log ${upstream}..${branchname}` ]]; then
      pushed="%{${fg[green]}%}[up]%{$reset_color%}"
    else
      pushed="%{${fg[red]}%}[up]%{$reset_color%}"
    fi
  fi

  echo "$branch$pushed"
}

# git-fzf.zsh
# 使い方:
#   source /workspaces/nova-parser/scripts/git-fzf.zsh
#   gbd-fzf           # マージ済みでなければ削除失敗（安全）
#   gbd-fzf -f        # 強制削除

function _gbd_fzf_default_base() {
  local base
  base=$(git symbolic-ref --quiet --short refs/remotes/origin/HEAD 2>/dev/null)
  if [[ -n "$base" ]]; then
    echo "$base"
    return 0
  fi

  if git show-ref --verify --quiet refs/remotes/origin/main; then
    echo "origin/main"
    return 0
  fi

  if git show-ref --verify --quiet refs/heads/main; then
    echo "main"
    return 0
  fi

  if git show-ref --verify --quiet refs/heads/master; then
    echo "master"
    return 0
  fi

  echo "HEAD"
}

function gbd-fzf() {
  emulate -L zsh
  setopt pipefail

  if ! command -v git >/dev/null 2>&1; then
    echo "git が見つかりません" >&2
    return 1
  fi

  if ! command -v fzf >/dev/null 2>&1; then
    echo "fzf が見つかりません" >&2
    return 1
  fi

  git rev-parse --is-inside-work-tree >/dev/null 2>&1 || {
    echo "git リポジトリ内で実行してください" >&2
    return 1
  }

  local delete_cmd="-d"
  if [[ "$1" == "-f" || "$1" == "--force" ]]; then
    delete_cmd="-D"
  fi

  local selected
  selected=$(git for-each-ref refs/heads --format='%(if)%(HEAD)%(then)* %(else)  %(end)%(refname:short)' \
    | fzf --height=40% --reverse --prompt='Delete local branch > ' --header='Enter: 選択 / ESC: キャンセル') || return 0

  local branch=${selected#\* }
  branch=${branch##  }

  if [[ "$selected" == \** ]]; then
    echo "現在のブランチは削除できません: $branch" >&2
    return 1
  fi

  local ans
  read "ans?ローカルブランチ '$branch' を削除しますか？ [y/N]: "
  if [[ "$ans" != "y" && "$ans" != "Y" ]]; then
    echo "キャンセルしました"
    return 0
  fi

  if [[ "$delete_cmd" == "-D" ]]; then
    git branch -D -- "$branch"
    return $?
  fi

  if git branch -d -- "$branch"; then
    return 0
  fi

  local base
  base=$(_gbd_fzf_default_base)

  local branch_tree
  local base_tree
  branch_tree=$(git rev-parse "$branch^{tree}" 2>/dev/null)
  base_tree=$(git rev-parse "$base^{tree}" 2>/dev/null)

  if [[ -n "$branch_tree" && -n "$base_tree" && "$branch_tree" == "$base_tree" ]]; then
    echo "注意: 祖先関係はありませんが、'$branch' は '$base' と内容が同一です。"
    local ans2
    read "ans2?内容同一のため強制削除しますか？ [y/N]: "
    if [[ "$ans2" == "y" || "$ans2" == "Y" ]]; then
      git branch -D -- "$branch"
      return $?
    fi
    echo "キャンセルしました"
    return 1
  fi

  echo "削除できませんでした。未マージの場合は gbd-fzf -f を使ってください。" >&2
  return 1
}
