function herdrstart(){
    if [ "${HERDR_ENV:-}" = "1" ] || [ -n "${TMUX:-}" ]; then
        return 0
    fi
    export TZ="${TZ:-Asia/Tokyo}"
    if ! command -v herdr >/dev/null 2>&1; then
        print -u2 "warning: herdr not found; continuing without a multiplexer"
        return 0
    fi
    herdr
}
