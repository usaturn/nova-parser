function git-prompt {
  local branchname branch st remote pushed upstream
  branchname=`git symbolic-ref --short HEAD 2> /dev/null`
  if [ -z $branchname ]; then
    return
  fi
  st=`git status 2> /dev/null`
  if [[ -n `echo "$st" | grep "^nothing to"` ]]; then
    branch="%{${fg[green]}%}($branchname)%{$reset_color%}"
  elif [[ -n `echo "$st" | grep "^nothing added"` ]]; then
    branch="%{${fg[yellow]}%}($branchname)%{$reset_color%}"
  else
    branch="%{${fg[red]}%}($branchname)%{$reset_color%}"
  fi

  remote=`git config branch.${branchname}.remote 2> /dev/null`

  if [ -z $remote ]; then
    pushed=''
  else
    upstream="${remote}/${branchname}"
    if [[ -z `git log ${upstream}..${branchname}` ]]; then
      pushed="%{${fg[green]}%}[up]%{$reset_color%}"
    else
      pushed="%{${fg[red]}%}[up]%{$reset_color%}"
    fi
  fi

  echo "$branch$pushed"
}

function awsenv(){
    export AWS_DEFAULT_PROFILE=$(AWS_CLI_AUTO_PROMPT="" aws configure list-profiles | fzf) &&
    AWS_PAGER="" aws sts get-caller-identity --output yaml &&
    AWS_PAGER="" aws iam list-account-aliases --query 'AccountAliases|[0]'

function jssenv(){
    local choice
    choice=$(printf "test-tokyo\ntest-osaka\njss-tokyo\njss-osaka\n" | fzf)
    if [[ -z "$choice" ]]; then
        echo "環境が選択されませんでした。" >&2 # エラーメッセージは標準エラー出力へ
        return 1 # 関数をエラーで終了
    fi

    export ENVIRONMENT_NAME="$choice"

    case "${ENVIRONMENT_NAME}" in
      jss-tokyo)
        export REGION="ap-northeast-1"
        export AWS_DEFAULT_PROFILE=jss-prod-tokyo
        ;;
      jss-osaka)
        export REGION="ap-northeast-3"
        export AWS_DEFAULT_PROFILE=jss-prod-osaka
        ;;
      test-tokyo)
        export REGION="ap-northeast-1"
        export AWS_DEFAULT_PROFILE=solution2-tokyo
        ;;
      test-osaka)
        export REGION="ap-northeast-3"
        export AWS_DEFAULT_PROFILE=solution2-osaka
        ;;
      *)
        echo "エラー: 予期しない環境名 '${ENVIRONMENT_NAME}' です。REGION は設定されません。" >&2
        return 1
        ;;
    esac

    AWS_PAGER="" aws sts get-caller-identity --output yaml
    AWS_PAGER="" aws iam list-account-aliases --query 'AccountAliases|[0]'
    echo "以下の環境変数が設定されました:"
    cat <<- EOF
          ENVIRONMENT_NAME    = ${ENVIRONMENT_NAME}
          REGION              = ${REGION}
          AWS_DEFAULT_PROFILE = ${AWS_DEFAULT_PROFILE}
EOF
    return 0
}

function ec2connect(){
    export INSTANCE_NAME=$(aws ec2 describe-instances --query 'Reservations[].Instances[?IamInstanceProfile!=`null` && State.Name == `running`].[Tags[?Key==`Name`] | [0].Value]' --output text | fzf)
    export INSTANCE_ID=$(aws ec2 describe-instances --filters "Name=tag:Name,Values=${INSTANCE_NAME}"  Name=instance-state-name,Values=running --query 'Reservations[].Instances[].InstanceId' --output text)
    if [ -n "${INSTANCE_ID}" ] ;then
        echo connect: ${INSTANCE_NAME}
    else
        echo INSTANCE_ID is none
        return 1
    fi
    aws ssm start-session --target ${INSTANCE_ID} || echo $?
}

function ec2start(){
    export INSTANCE_NAME=$(aws ec2 describe-instances --query 'Reservations[].Instances[?IamInstanceProfile!=`null` && State.Name==`stopped`].[Tags[?Key==`Name`] | [0].Value]' --output text | fzf)

    if [ -n "${INSTANCE_ID}" ] ;then
        echo Start: ${INSTANCE_NAME}
    else
        echo INSTANCE_ID is none
        return 1
    fi
    aws ec2 start-instances --instance-ids ${INSTANCE_ID} || echo $?
    aws ec2 describe-instances --query 'Reservations[].Instances[].{
        Name:Tags[?Key==`Name`] | [0].Value,
        InstanceId:InstanceId,
        Status:State.Name,
        PublicIpAddress:PublicIpAddress,
        PrivateIpAddress:PrivateIpAddress,
        Az:Placement.AvailabilityZone}' --output table
}

function regist-target(){
    if [ -z "$AWS_DEFAULT_PROFILE" ]; then
        echo "Error: AWS_DEFAULT_PROFILE environment variable is not set"
        echo "Execute awsenv!"
        awsenv
    fi
    export INSTANCE_NAME=$(aws ec2 describe-instances --query 'Reservations[].Instances[?IamInstanceProfile!=`null`].[Tags[?Key==`Name`] | [0].Value]' --output text | fzf)
    export INSTANCE_ID=$(aws ec2 describe-instances --filters "Name=tag:Name,Values=${INSTANCE_NAME}" --query 'Reservations[].Instances[].InstanceId' --output text)
    INSTANCE_STATE=$(aws ec2 describe-instances --instance-ids ${INSTANCE_ID} --query 'Reservations[].Instances[].State.Name' --output text)
    if [ "${INSTANCE_STATE}" != "running" ]; then
        echo "Instance ${INSTANCE_NAME} (${INSTANCE_ID}) is ${INSTANCE_STATE}. Starting instance..."
        aws ec2 start-instances --instance-ids ${INSTANCE_ID}
        echo "Waiting for instance to be running..."
        aws ec2 wait instance-running --instance-ids ${INSTANCE_ID}
        echo "Instance is now running"
    else
        echo "Instance ${INSTANCE_NAME} (${INSTANCE_ID}) is already running"
    fi
    if [ -n "${INSTANCE_ID}" ] ;then
        export LOAD_BALANCER_ARN=$(aws elbv2 describe-load-balancers --query 'LoadBalancers[].[LoadBalancerArn]' --output text|fzf) && echo ${LOAD_BALANCER_ARN}
        export TARGET_GROUP_NAME=$(aws elbv2 describe-target-groups --load-balancer-arn=${LOAD_BALANCER_ARN} --query 'TargetGroups[].TargetGroupName'  --output text) && echo ${TARGET_GROUP_NAME}
        export TARGET_GROUP_ARN=$(aws elbv2 describe-target-groups --names ${TARGET_GROUP_NAME} --query 'TargetGroups[].TargetGroupArn' --output text) && echo ${TARGET_GROUP_ARN}
        aws elbv2 register-targets --target-group-arn ${TARGET_GROUP_ARN} --targets Id=${INSTANCE_ID}
    else
        echo INSTANCE_ID is none
        return 1
    fi
}
